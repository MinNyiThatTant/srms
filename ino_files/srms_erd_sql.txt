Project SRMS_DB {
  database_type: "MySQL"
}

Table main_categories {
  id int [pk, increment]
  name varchar(100) [unique, not null]
}

Table sub_categories {
  id int [pk, increment]
  main_category_id int [not null]
  name varchar(100) [not null]
  created_at timestamp

  indexes {
    (main_category_id, name) [unique]
  }
}

Table menu {
  id int [pk, increment]
  name varchar(255)
  price int
  image varchar(255) [default: 'default.jpg']
  category varchar(100)
  main_category varchar(100)
  sub_category_id int
  main_category_id int
}

Table orders {
  id int [pk, increment]
  table_no varchar(50)
  total_price int
  status varchar(50) [default: 'pending']
  created_at timestamp
}

Table order_details {
  id int [pk, increment]
  order_id int
  item_name varchar(255)
  price int
}

Table categories {
  id int [pk, increment]
  main_category varchar(100) [not null]
  sub_category_name varchar(100) [not null]
}

/* =========================
   RELATIONSHIPS
========================= */

Ref: sub_categories.main_category_id > main_categories.id

Ref: menu.sub_category_id > sub_categories.id

Ref: menu.main_category_id > main_categories.id

Ref: order_details.order_id > orders.id

/*
Optional Logical Relationship
(categories table uses string relation)
*/

Ref: categories.main_category > main_categories.name
